 30:     {

 84:     {
 85:         if (!is_array($array)) {

123:     /**
124:      * Finds intersection between two (endless) lines
125:      *
126:      * @param Point $point1 The id of the start of line A
127:      * @param Point $point2 The id of the end line A
128:      * @param Point $point3 The id of the start of line B
129:      * @param Point $point4 The id of the end line B
130:      *
131:      * @return array|null The coordinates of the intersection, or null if the lines are parallel
132:      */
133:     public static function lineLineIntersection(Point $point1, Point $point2, Point $point3, Point $point4)
134:     {
135:         /* weed out parallel lines */
136:         if ($point1->getX() == $point2->getX() && $point3->getX() == $point4->getX()) {
137:             return false;
138:         }
139:         if ($point1->getY() == $point2->getY() && $point3->getY() == $point4->getY()) {
140:             return false;
141:         }
142: 
143:         /* If line is vertical, handle this special case */
144:         if ($point1->getX() == $point2->getX()) {
145:             $slope = Utils::getSlope($point3, $point4);
146:             $i = $point3->getY() - ($slope * $point3->getX());
147:             $x = $point1->getX();
148:             $y = $slope * $x + $i;
149:         } elseif ($point3->getX() == $point4->getX()) {
150:             $slope = Utils::getSlope($point1, $point2);
151:             $i = $point1->getY() - ($slope * $point1->getX());
152:             $x = $point3->getX();
153:             $y = $slope * $x + $i;
154:         } else {
155:             /* If line goes from right to left, swap points */
156:             if ($point1->getX() > $point2->getX()) {
157:                 $tmp = $point1;
158:                 $point1 = $point2;
159:                 $point2 = $tmp;
160:             }
161:             if ($point3->getX() > $point4->getX()) {
162:                 $tmp = $point3;
163:                 $point3 = $point4;
164:                 $point4 = $tmp;
165:             }
166:             /* Find slope */
167:             $slope1 = Utils::getSlope($point1, $point2);
168:             $slope2 = Utils::getSlope($point3, $point4);
169:             /* Find y intercept */
170:             $i1 = $point1->getY() - ($slope1 * $point1->getX());
171:             $i2 = $point3->getY() - ($slope2 * $point3->getX());
172:             /* Find intersection */
173:             $x = ($i2 - $i1) / ($slope1 - $slope2);
174:             $y = $slope1 * $x + $i1;
175:         }
176: 
177:         return [$x, $y];
178:     }
