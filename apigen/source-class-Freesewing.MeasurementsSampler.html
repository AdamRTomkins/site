 27:      *
 28:      * @param string group Name of the model group as defined in sampler config
 29:      *
 30:      * @return array|null Array of models or null if we can't read the config file

 65:     /**
 66:      * Samples the pattern for each model (set of measurements)
 67:      *
 68:      * For each model, this clones the pattern and calls the sample() method
 69:      * with the model as parameter.
 70:      * It then itterates over the parts and calls sampleParts() on them
 71:      *
 72:      * @param \Freesewing\Themes\Theme or similar
 73:      *
 74:      * @return \Freesewing\Patterns\Pattern or similar
 75:      */
 76:     public function sampleMeasurements($theme)
 77:     {
 78:         $renderBot = new \Freesewing\SvgRenderbot();
 79:         $steps = count($this->models);
 80:         $i = 0;
 81:         foreach ($this->models as $modelKey => $model) {
 82:             if($modelKey == 'compareModel') $mode = 'compare';
 83:             else $mode = 'sample';
 84:             $p = clone $this->pattern;
 85:             $p->loadParts();
 86:             $p->sample($model);
 87:             foreach ($p->parts as $partKey => $part) {
 88:                 $this->sampleParts($i, $steps, $p, $theme, $renderBot, $mode);
 89:             }
 90:             ++$i;
 91:         }
 92:         $this->addSampledPartsToPattern();
 93:         $theme->applyRenderMask($this->pattern);
 94:         $this->pattern->layout();
 95:         return $this->pattern;
 96:     }

101:      * @param string $group Name of the model group as defined in sampler config
102:      *
103:      * @return array Array of models
104:      */
105:     private function loadModelGroup($group)
106:     {
107:         $models = [];

109:             $model = new \Freesewing\Model();

113:             $model->addMeasurements($measurements);

116:         }
