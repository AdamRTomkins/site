 48:     /**
 49:      * Returns the anchor point to be used for sampling
 50:      *
 51:      * This will check for:
 52:      *  - a point with id 'samplerAnchor'
 53:      *  - a point with id 'gridAnchor'
 54:      * It will return the first one found.
 55:      * If the part has neither, it will add a point
 56:      * with coordinates (0,0) and return that.
 57:      *
 58:      * @param \Freesewing\Part $part The part
 59:      *
 60:      * @return \Freesewing\Point The anchor point
 61:      */
 62:     private function getSamplerAnchor($part)
 63:     {
 64:         if (isset($part->points['samplerAnchor'])) {
 65:             return $part->loadPoint('samplerAnchor');
 66:         } elseif (isset($part->points['gridAnchor'])) {
 67:             return $part->loadPoint('gridAnchor');
 68:         } else {
 69:             $part->newPoint('samplerAnchor', 0, 0, 'Anchor point added by sampler');
 70:             return $part->loadPoint('samplerAnchor');
 71:         }
 72:     }

 93:     /**
 94:      * Samples parts for a given model and options
 95:      *
 96:      * The MeasurementsSampler iterates over models
 97:      * The OptionsSampler iterates over an option value
 98:      * When doing so, both call this function to do the actual sampling.
 99:      * It does two separate things:
100:      *  - It renders the paths marked for sampling
101:      *  - It finds a bounding box for the parts
102:      * This info is stored in $this->partContainer and will be added to a pattern later
103:      * The $step and $steps parameters are passed to the theme so different samplings can be
104:      * made to look different. (giving us that rainbow effect in the standard theme).
105:      *
106:      * @param int                      $step      The step out of total steps this is
107:      * @param int                      $step      The total amount of steps
108:      * @param \Freesewing\Pattern      $pattern   The pattern to sample
109:      * @param \Freesewing\Theme        $theme     The theme
110:      * @param \Freesewing\SvgRenderbot $mode      sample or compare
111:      * @param \Freesewing\SvgRenderbot $renderBot The SVG renderbot to render the path
112:      * @param string                   $mode      sample or compare
113:      */
114:     protected function sampleParts($step, $steps, $pattern, $theme, $renderBot, $mode='sample')
115:     {
116:         foreach ($pattern->parts as $partKey => $part) {
117:             if ($part->getRender() === true) {
118:                 if (!isset($this->anchors[$partKey])) {
119:                     $this->anchors[$partKey] = $this->getSamplerAnchor($part);
120:                     $deltaX = 0;
121:                     $deltaY = 0;
122:                     $transform = 'translate( 0, 0 )';
123:                 } else {
124:                     $anchor = $this->getSamplerAnchor($part);
125:                     $deltaX = $this->anchors[$partKey]->getX() - $anchor->getX();
126:                     $deltaY = $this->anchors[$partKey]->getY() - $anchor->getY();
127:                     $transform = "translate( $deltaX, $deltaY )";
128:                 }
129:                 foreach ($part->paths as $pathKey => $path) {
130:                     if ($path->getSample() === true) {
131:                         $path->boundary = $path->findBoundary($part);
132:                         if (!isset($this->boundaries[$partKey]) || !is_object($this->boundaries[$partKey]['topLeft'])) {
133:                             $this->boundaries[$partKey]['topLeft'] = new \Freesewing\Point();
134:                             $this->boundaries[$partKey]['topLeft']->setX($path->boundary->topLeft->x);
135:                             $this->boundaries[$partKey]['topLeft']->setY($path->boundary->topLeft->y);
136:                             $this->boundaries[$partKey]['bottomRight'] = new \Freesewing\Point();
137:                             $this->boundaries[$partKey]['bottomRight']->setX($path->boundary->bottomRight->x);
138:                             $this->boundaries[$partKey]['bottomRight']->setY($path->boundary->bottomRight->y);
139:                         } else {
140:                             if (($path->boundary->topLeft->x + $deltaX) < $this->boundaries[$partKey]['topLeft']->x) {
141:                                 $this->boundaries[$partKey]['topLeft']->setX($path->boundary->topLeft->x + $deltaX);
142:                             }
143:                             if (($path->boundary->topLeft->y + $deltaY) < $this->boundaries[$partKey]['topLeft']->y) {
144:                                 $this->boundaries[$partKey]['topLeft']->setY($path->boundary->topLeft->y + $deltaY);
145:                             }
146:                             if ($path->boundary->bottomRight->x + $deltaX > $this->boundaries[$partKey]['bottomRight']->x) {
147:                                 $this->boundaries[$partKey]['bottomRight']->setX($path->boundary->bottomRight->x + $deltaX);
148:                             }
149:                             if ($path->boundary->bottomRight->y + $deltaY > $this->boundaries[$partKey]['bottomRight']->y) {
150:                                 $this->boundaries[$partKey]['bottomRight']->setY($path->boundary->bottomRight->y + $deltaY);
151:                             }
152:                         }
153:                         if($mode == 'compare') $path->setAttributes(['transform' => $transform, 'class' => 'compare']);
154:                         else $path->setAttributes(['transform' => $transform, 'style' => $theme->samplerPathStyle($step, $steps)]);
155:                         $this->partContainer[$partKey]['includes']["$step-$pathKey"] = $renderBot->renderPath($path, $part);
156:                         $this->partContainer[$partKey]['topLeft'] = $this->boundaries[$partKey]['topLeft'];
157:                         $this->partContainer[$partKey]['bottomRight'] = $this->boundaries[$partKey]['bottomRight'];
158:                     }
159:                 }
160:             }
161:         }
162:     }

164:     /**
165:      * Adds all sampled parts to the pattern
166:      *
167:      * This stored the parts in $this->partContainer in the pattern
168:      * but not before drawing a path in them along the boundary.
169:      * That's needed because the sampled paths are stored as rendered snippets
170:      * and thus we can no longer determine their boundary.
171:      */
172:     protected function addSampledPartsToPattern()
173:     {
174:         foreach ($this->partContainer as $partKey => $part) {
175:             $this->pattern->newPart("sampler-$partKey");
176:             $p = $this->pattern->parts[$partKey];
177:             $p->newPoint(1, $part['topLeft']->getX(), $part['topLeft']->getY(), 'Top left');
178:             $p->newPoint(3, $part['bottomRight']->getX(), $part['bottomRight']->getY(), 'Bottom right');
179:             $p->newPoint(2, $p->x(3), $p->y(1), 'Top right');
180:             $p->newPoint(4, $p->x(1), $p->y(3), 'Bottom left');
181:             $p->newPath('border', 'M 1 L 2 L 3 L 4 z', ['class' => 'hidden']);
182:             foreach ($part['includes'] as $pathKey => $include) {
183:                 $p->newInclude($pathKey, $include);
184:             }
185:         }
186:     }
