175:     /**
176:      * Loads templates from themes and possible parent themes
177:      *
178:      * This makes sure that when you extend a theme, the templates
179:      * are extended too.
180:      * That means that the parent theme templates are loaded, unless
181:      * you override them in your extended theme.
182:      *
183:      * @return array $templates Array with template files
184:      */
185:     private function loadTemplateHierarchy()
186:     {
187:         $locations = $this->getClassChain();
188:         $templates = array();
189:         foreach ($locations as $location) {
190:             if (is_readable("$location/config.yml")) {
191:                 $dir = "$location/templates";
192:                 $config = \Freesewing\Yamlr::loadYamlFile("$location/config.yml");
193:                 if (isset($config['templates'])) {
194:                     foreach ($config['templates'] as $type => $entries) {
195:                         foreach ($entries as $entry) {
196:                             if (!isset($templates[$type][$entry])) {
197:                                 $template = "$location/templates/$entry";
198:                                 if (is_readable($template)) {
199:                                     $templates[$type][$entry] = file_get_contents($template);
200:                                 }
201:                             }
202:                         }
203:                     }
204:                 }
205:             }
206:         }
207: 
208:         return $templates;
209:     }

211:     /**
212:      * Returns an array of classes between this and the abstract theme class
213:      *
214:      * If your ptheme extends another theme (and another, and ...) this
215:      * will construct the class chain up to the abstract theme class.
216:      * This is needed for the theme to load template files hierarchically
217:      * In other words, when you extend a theme, this makes sure the templates
218:      * are extended too.
219:      *
220:      * @return array $locations An array of class directories
221:      */
222:     private function getClassChain()
223:     {
224:         $reflector = new \ReflectionClass(get_class($this));
225:         $filename = $reflector->getFileName();
226:         $locations[] = dirname($filename);
227:         do {
228:             $parent = $reflector->getParentClass();
229:             $reflector = new \ReflectionClass($parent->name);
230:             $filename = $reflector->getFileName();
231:             $locations[] = dirname($filename);
232:         } while ($parent->name != 'Freesewing\Themes\Core\Theme');
233: 
234:         return $locations;
235:     }

300:     /**
301:      * Sets the render property on parts based on theme options
302:      *
303:      * You can use the following request parameter to control the render mask for parts:
304:      *
305:      *  - parts
306:      *  - forceParts
307:      *
308:      * If *parts* is an array, all parts NOT in the array will NOT be rendered
309:      * If in addition, *forceParts* is 1, only parts in the *parts* array will be rendered
310:      * The difference is that *forceParts* will force all parts in the *parts* array
311:      * to be rendered. Even those that have their render property set to false.
312:      *
313:      * @param \Freesewing\Patterns\* $pattern The pattern object
314:      */
315:     private function applyRenderMaskOnParts($pattern)
316:     {
317:         $parts = $this->getOption('parts');
318:         // Force into array, even if it's just 1 part
319:         if(isset($parts) && !is_array($parts) && isset($pattern->parts[$parts])) $parts = [$parts];
320:         
321:         foreach ($pattern->parts as $key => $part) {
322:             if (!in_array($key, $parts)) {
323:                 // Don't render what's not included
324:                 $pattern->parts[$key]->setRender(false);
325:             } else {
326:                 if ($this->getOption('forceParts')) {
327:                     // Force render of what's included
328:                     $pattern->parts[$key]->setRender(true);
329:                 }
330:             }
331:         }
